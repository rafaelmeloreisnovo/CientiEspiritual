def codepoint_info(cp):
    ch = chr(cp)
    name = unicodedata.name(ch, "<unassigned>")
    nfkc = unicodedata.normalize('NFKC', ch)
    return {"cp": cp, "hex": f"U+{cp:04X}", "char": ch, "name": name, "nfkc": nfkc}
